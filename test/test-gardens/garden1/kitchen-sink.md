### Python

```python
def greet(name):
    """Greet someone by name."""
    return f"Hello, {name}!"

def fibonacci(n):
    """Generate fibonacci sequence up to n."""
    a, b = 0, 1
    while a < n:
        yield a
        a, b = b, a + b

# Example usage
print(greet("World"))
for num in fibonacci(100):
    print(num, end=' ')
```
